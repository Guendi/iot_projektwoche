# DVD-Support (besides VLC)
wget http://www.videolan.org/pub/libdvdcss/1.2.12/libdvdcss-1.2.12.tar.bz2
bzip2 -d libdvdcss-1.2.12.tar.bz2
tar -xvf libdvdcss-1.2.12.tar

cd libdvdcss-1.2.12
./configure
make
sudo make install
sudo cp /usr/local/lib/libdvdcss* /lib
